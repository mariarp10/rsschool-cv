  • The function finds the unique number in an array where all other numbers appear more than once. 
                  
function findUniq(arr) {
const counter = new Map();
let unique;
for (const num of arr) {
  if (!counter.has(num)) {
    counter.set(num, 1);
  } else {
    counter.set(num, counter.get(num) + 1);
  }
}
for (const [key, value] of counter.entries()) {
  if (value === 1) {
    unique = key;
  }
}
return unique;
}

const sqrtCache = new Map();                 
function solveQuadratic(a, b, c) {
if (a === 0) {
if (b === 0) return;
  return [-c / b];
}
const D = b ** 2 - 4 * a * c;                 
if (D < 0) return;
  if (D === 0) return [-b / (2 * a)];                 
let sqrtD;
if (sqrtCache.has(D)) {
  sqrtD = sqrtCache.get(D);
} else {
  sqrtD = Math.sqrt(D);
  sqrtCache.set(D, sqrtD);
}                 
return [(-b + sqrtD) / (2 * a), (-b - sqrtD) / (2 * a)];
}
